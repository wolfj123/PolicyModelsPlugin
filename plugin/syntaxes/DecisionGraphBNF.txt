<free_text>				::=  (<letter> <digit>  .,/~?!()@:#$%^&*_+-  )+

<node_id> 				::= <letter> (<letter> <digit>  .,/~?!()@:#$%^&*_+-  )*	

<letter> 					::= "a" | "b" | ... | "z" | "A" | ... | "Z"

<digit> 					::= "0" | "1" | "2" | ... | "9"
